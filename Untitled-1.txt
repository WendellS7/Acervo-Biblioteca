Conta * obterConta(vector<Conta *> listContas){
	
	string agencia, numConta;
	Conta *conta = NULL;
	
	cout << "Entre com a ag�ncia: ";
	getline(cin, agencia);
	cin.clear();
	fflush(stdin);
	
	cout << "Entre com o n�mero da conta: ";
	getline(cin, numConta);
	cin.clear();
	fflush(stdin);
	
	for(vector<Conta *>::iterator it = listContas.begin(); it != listContas.end(); it ++){	
		Conta *c = (*it);
		if(c->getAgencia().compare(agencia) == 0 && c->getNumConta().compare(numConta) == 0){
			conta = c;
			break;
		}
		
	}
	
	return conta;	
}
